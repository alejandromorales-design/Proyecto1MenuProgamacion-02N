package co.edu.uniquindio.biblioteca.model;

public class Libro {
    public String nombre;
    public int codigo;
    public String genero;
    public String idioma;

    public Libro(String nombre, int codigo, String genero, String idioma) {
        this.nombre = nombre;
        this.codigo = codigo;
        this.genero = genero;
        this.idioma = idioma;
    }
    public Libro() {
    }
    public String getNombre() {
        return nombre;
    }

    public static void setNombre(String nombreLibro) {
    }
    public int getCodigo() {
        return codigo;
    }

    public static void setCodigo(int codigoLibro) {
    }
    public String getGenero() {
        return genero;
    }

    public static void setGenero(String generoLibro) {
    }
    public String getIdioma() {
        return idioma;
    }

    public static void setIdioma(String idiomaLibro) {
    }
}
