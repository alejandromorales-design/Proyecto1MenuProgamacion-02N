package co.edu.uniquindio.biblioteca;

import co.edu.uniquindio.biblioteca.model.Cliente;

import co.edu.uniquindio.biblioteca.model.Empleado;
import co.edu.uniquindio.biblioteca.model.Libro;

import java.util.Scanner;


public class Main {
    public static void main(String[] args) {
        mostrarMenu();

        int opcionSeleecionada= leerEntero("Ingrese un número de acuerdo a las opciones");
        if(opcionSeleecionada==1){
            String respuesta = crearCliente();
            System.out.println(respuesta + "Cliente creado");
        } else if(opcionSeleecionada==2){
            String respuesta = crearEmpleado();
            System.out.println(respuesta + "Empleado creado");
        } else if (opcionSeleecionada==3) {
            String respuesta = seleccionarLibro();
            System.out.println(respuesta + "Libro Seleccionado");
        } else if (opcionSeleecionada==4) {
            System.out.println("Saliendo de la aplicacion");

        }else{ System.out.println("Opcion no es valida");


        }
    }

    public static String crearCliente(){
        String nombreCliente = leerString("Ingrese el nombre del cliennte: ");
        String apellidoCliente = leerString("Ingrese el apellido del cliente: ");
        int idCliente = leerEntero("Ingrese el ID del cliente: ");
        int edadCliente= leerEntero("Ingrese la edad del cliente: ");
            if ( edadCliente<10 || edadCliente>100){
                System.out.println("La edad del cliente es invalida");
            }
        int telefonoCliente = leerEntero("Ingrese el número de telefono del cliente");

        Cliente cliente = new Cliente(nombreCliente, apellidoCliente, idCliente, edadCliente, telefonoCliente);
        cliente.setNombre(nombreCliente);
        cliente.setApellido(apellidoCliente);
        cliente.setId(idCliente);
        cliente.setEdad(edadCliente);
        cliente.setTelefono(telefonoCliente);
        return cliente.toString();

    }
public static String crearEmpleado(){
        String nombreEmpleado =  leerString("Ingrese el nombre del empleado");
        String apellidoEmpleado = leerString("Ingrese el apellido del empleado");
        int cedulaEmpleado = leerEntero("Ingrese el cedula del empleado");
        int edadEmpleado=  leerEntero("Ingrese la edad del empleado");

        Empleado TEMPORALempleado = new Empleado(nombreEmpleado,apellidoEmpleado, cedulaEmpleado, edadEmpleado);
        TEMPORALempleado.setNombre(nombreEmpleado);
        TEMPORALempleado.setApellido(apellidoEmpleado);
        TEMPORALempleado.setCedula(cedulaEmpleado);
        TEMPORALempleado.setEdad(edadEmpleado);
    return nombreEmpleado;
}

public static String seleccionarLibro(){
        String nombreLibro = leerString("Ingrese el nombre del libro: ");
        int codigoLibro=  leerEntero("Ingrese el codigo del libro: ");
        String generoLibro =  leerString("Ingrese el genero del libro: ");
        String idiomaLibro =  leerString("Ingrese el idioma libro: ");

    Libro libro = new Libro(nombreLibro, codigoLibro, generoLibro, idiomaLibro);

    libro.setNombre(nombreLibro);
    libro.setCodigo(codigoLibro);
    libro.setGenero(generoLibro);
    libro.setIdioma(idiomaLibro);
    return libro.toString();


}

public static String leerString(String mensaje){
        String respuesta = "";
        System.out.print(mensaje);
        Scanner leer = new Scanner(System.in);
        respuesta = leer.nextLine();
        return respuesta;
}

public static int leerEntero(String mensaje){
        int respuesta = 0;
        System.out.print(mensaje);
        Scanner leer = new Scanner(System.in);
        respuesta = leer.nextInt();
        return respuesta;
}

public static void mostrarMenu(){
        System.out.println("1/ Crear cliente");
        System.out.println("2/ Crear empleado");
        System.out.println("3/ Seleccionar libro");
        System.out.println("4/ Salir");
    }

}
